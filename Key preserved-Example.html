DROP TABLE Employees;
DROP TABLE Employees2;
DROP TABLE Departments;

-- KEY-PRESERVED EXAMPLE

-- 1. Create Departments table
CREATE TABLE Departments (
    dept_id NUMBER PRIMARY KEY,
    department_name VARCHAR2(50)
);

-- 2. Create Employees table
CREATE TABLE Employees (
    emp_id NUMBER PRIMARY KEY,
    emp_name VARCHAR2(50),
    dept_id NUMBER REFERENCES Departments(dept_id)
);

-- 3. Insert Departments data
INSERT INTO Departments VALUES (1, 'HR');
INSERT INTO Departments VALUES (2, 'Finance');
INSERT INTO Departments VALUES (3, 'IT');

-- 4. Insert Employees data (1 employee per department)
INSERT INTO Employees VALUES (1, 'John', 1);
INSERT INTO Employees VALUES (2, 'Mary', 2);
INSERT INTO Employees VALUES (3, 'Mike', 3);

-- 5. Join Query (Key-Preserved)
SELECT e.emp_id, e.emp_name, d.dept_id, d.department_name
FROM Employees e
JOIN Departments d ON e.dept_id = d.dept_id;

-- NOT KEY-PRESERVED EXAMPLE

-- 1. (Use same Departments table as before)

-- 2. Create Employees2 table
CREATE TABLE Employees2 (
    emp_id NUMBER PRIMARY KEY,
    emp_name VARCHAR2(50),
    dept_id NUMBER REFERENCES Departments(dept_id)
);

-- 3. Insert Employees2 data (two employees in dept 1)
INSERT INTO Employees2 VALUES (1, 'John', 1);
INSERT INTO Employees2 VALUES (2, 'Sam', 1);
INSERT INTO Employees2 VALUES (3, 'Mary', 2);

-- 4. Join Query (Not Key-Preserved)
SELECT e.emp_id, e.emp_name, d.dept_id, d.department_name
FROM Employees2 e
JOIN Departments d ON e.dept_id = d.dept_id;
