-- create client table similar to branch table
CREATE TABLE clients (
client_id NUMBER PRIMARY KEY,
location VARCHAR(30) NOT NULL,
addr VARCHAR(200) );

CREATE TABLE branch (
    branch_id NUMBER PRIMARY KEY,
    location  VARCHAR2(30) NOT NULL,
    addr      VARCHAR2(200)
);

INSERT INTO branch (branch_id, location, addr)
VALUES (1, 'NewYork', '25 10th Block');

INSERT INTO branch (branch_id, location, addr)
VALUES (2, 'Canada', '120 15th Block');


-- combines the two tables removing duplicates
SELECT * FROM branch
UNION 
SELECT * FROM clients;

-- combines the two tables without removing duplicates
SELECT * FROM branch
UNION ALL
SELECT * FROM clients;
